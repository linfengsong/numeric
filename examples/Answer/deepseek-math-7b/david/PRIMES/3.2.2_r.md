theorem primes_between_100_and_200 : 
  {p | 100 ≤ p ∧ p ≤ 200 ∧ Nat.Prime p} = 
  {101, 103, 107, 109, 113, 127, 131, 137, 139, 149, 151, 157, 163, 167, 173, 179, 181, 191, 193, 197, 199} := by
  ext p
  simp only [Set.mem_setOf_eq, Set.mem_insert_iff, Set.mem_singleton_iff]
  constructor
  -- First, we prove that if p is a prime number between 100 and 200, then p is in the given set.
  · rintro ⟨h₁, h₂, h₃⟩
    have h₄ : p ≥ 100 := h₁
    have h₅ : p ≤ 200 := h₂
    have h₆ : Nat.Prime p := h₃
    -- We use a case analysis on the possible values of p to show it must be in the set.
    interval_cases p <;> norm_num [Nat.Prime] at h₆ <;> aesop
  -- Next, we prove that if p is in the given set, then p is a prime number between 100 and 200.
  · rintro (rfl | rfl | rfl | rfl | rfl | rfl | rfl | rfl | rfl | rfl | rfl | rfl | rfl | rfl | rfl | rfl | rfl | rfl | rfl | rfl | rfl) <;>
    norm_num [Nat.Prime]
