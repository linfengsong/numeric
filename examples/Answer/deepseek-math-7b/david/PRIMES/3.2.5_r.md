theorem any_composite_three_digit_number_has_prime_factor_le_31 : ∀ n : ℕ, 100 ≤ n → n ≤ 999 → ¬ Nat.Prime n → ∃ p : ℕ, p.Prime ∧ p ≤ 31 ∧ p ∣ n := by
  intro n h₁ h₂ h₃
  have h₄ : n ≥ 100 := by linarith
  have h₅ : n ≤ 999 := by linarith
  have h₆ : n ≥ 2 := by linarith
  -- We use the fact that if a number is not prime, it has a prime factor less than or equal to its square root.
  -- For three-digit numbers, the square root is less than 32, so we only need to check primes up to 31.
  have h₇ : ∃ p : ℕ, p.Prime ∧ p ∣ n := by
    apply Nat.exists_prime_and_dvd
    linarith
  obtain ⟨p, hp, hpdvd⟩ := h₇
  use p
  constructor
  exact hp
  constructor
  -- We need to show that p ≤ 31.
  -- Since p is a prime factor of n, and n is a three-digit number, p must be less than or equal to the square root of 999, which is approximately 31.6.
  have h₈ : p ≤ 31 := by
    by_contra h
    have h₉ : p ≥ 32 := by linarith
    have h₁₀ : p ∣ n := by exact hpdvd
    have h₁₁ : p ≤ n := by
      apply Nat.le_of_dvd
      linarith
      exact h₁₀
    have h₁₂ : n < 1000 := by linarith
    have h₁₃ : p < 1000 := by linarith
    interval_cases p <;> norm_num [Nat.Prime] at hp <;> omega
  exact h₈
  exact hpdvd
