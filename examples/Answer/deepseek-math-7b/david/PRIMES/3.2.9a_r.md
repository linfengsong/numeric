theorem exists_prime_lt_factorial_gt (n : ℕ) (hn : n > 2) : ∃ p, p.Prime ∧ n < p ∧ p < n! := by
  have h₁ : ∃ p, p.Prime ∧ n < p ∧ p ≤ n! := by
    -- Use Bertrand's Postulate to find a prime p such that n < p ≤ n!
    have h₂ : ∃ p, p.Prime ∧ n < p ∧ p ≤ n! := by
      apply Nat.exists_prime_lt_and_le_two_mul (n) (by
        -- Prove that n < 2 * n for n > 2
        have h₃ : n < 2 * n := by linarith
        -- Use the fact that n < 2 * n to show that n < n!
        have h₄ : n < n! := by
          -- Use the fact that n! is the product of all numbers from 1 to n
          -- and since n > 2, n! > n
          have h₅ : n! > n := by
            -- Use the fact that n! is the product of all numbers from 1 to n
            -- and since n > 2, n! > n
            have h₆ : ∀ k : ℕ, 2 ≤ k → k < k! := by
              intro k hk
              induction' hk with k hk ih
              · simp
              · simp_all [Nat.factorial_succ, Nat.mul_comm]
                <;> nlinarith
            have h₇ : n < n! := by
              apply h₆
              linarith
            exact h₇
          linarith
        exact h₄)
    exact h₂
  obtain ⟨p, hp₁, hp₂, hp₃⟩ := h₁
  use p
  constructor
  exact hp₁
  constructor
  exact hp₂
  -- Since p ≤ n!, we need to show that p < n!
  have h₄ : p < n! := by
    -- Since p ≤ n! and p ≠ n!, we have p < n!
    have h₅ : p ≤ n! := hp₃
    have h₆ : p ≠ n! := by
      intro h
      rw [h] at hp₁
      -- If p = n!, then p would be a prime factor of n!, which is impossible for n > 2
      have h₇ : n! < n! := by
        nlinarith [Nat.Prime.one_lt hp₁]
      linarith
    -- Since p ≤ n! and p ≠ n!, we have p < n!
    omega
  exact h₄
