theorem gcd_divides_b (a b : ℤ) (h : gcd (2 * a - 3 * b) (4 * a - 5 * b) ∣ b) : gcd (2 * a + 3) (4 * a + 5) = 1 := by
  have h₁ : gcd (2 * a - 3 * b) (4 * a - 5 * b) ∣ b := h
  have h₂ : gcd (2 * a - 3 * b) (4 * a - 5 * b) ∣ 2 * a - 3 * b := gcd_dvd_left (2 * a - 3 * b) (4 * a - 5 * b)
  have h₃ : gcd (2 * a - 3 * b) (4 * a - 5 * b) ∣ 4 * a - 5 * b := gcd_dvd_right (2 * a - 3 * b) (4 * a - 5 * b)
  have h₄ : gcd (2 * a - 3 * b) (4 * a - 5 * b) ∣ 4 * a - 5 * b := gcd_dvd_right (2 * a - 3 * b) (4 * a - 5 * b)
  have h₅ : gcd (2 * a - 3 * b) (4 * a - 5 * b) ∣ 2 * a - 3 * b := gcd_dvd_left (2 * a - 3 * b) (4 * a - 5 * b)
  have h₆ : gcd (2 * a - 3 * b) (4 * a - 5 * b) ∣ 4 * a - 5 * b := gcd_dvd_right (2 * a - 3 * b) (4 * a - 5 * b)
  have h₇ : gcd (2 * a - 3 * b) (4 * a - 5 * b) ∣ b := h₁
  have h₈ : gcd (2 * a - 3 * b) (4 * a - 5 * b) ∣ 2 * a - 3 * b := gcd_dvd_left (2 * a - 3 * b) (4 * a - 5 * b)
  have h₉ : gcd (2 * a - 3 * b) (4 * a - 5 * b) ∣ 4 * a - 5 * b := gcd_dvd_right (2 * a - 3 * b) (4 * a - 5 * b)
  have h₁₀ : gcd (2 * a - 3 * b) (4 * a - 5 * b) ∣ 2 * a - 3 * b := gcd_dvd_left (2 * a - 3 * b) (4 * a - 5 * b)
  have h₁₁ : gcd (2 * a - 3 * b) (4 * a - 5 * b) ∣ 4 * a - 5 * b := gcd_dvd_right (2 * a - 3 * b) (4 * a - 5 * b)
  -- Use the fact that gcd(2a - 3b, 4a - 5b) divides b to show that gcd(2a + 3, 4a + 5) = 1
  have h₁₂ : gcd (2 * a + 3) (4 * a + 5) = 1 := by
    -- Use the Euclidean algorithm to find the gcd of 2a + 3 and 4a + 5
    have h₁₃ : gcd (2 * a + 3) (4 * a + 5) = gcd (2 * a + 3) (2 * a + 2) := by
      -- Subtract twice the first number from the second number
      have h₁₄ : 4 * a + 5 = 2 * (2 * a + 3) + 1 := by ring
      rw [h₁₄]
      simp [gcd_comm]
      <;> simp_all [gcd_comm]
      <;> omega
    rw [h₁₃]
    -- Continue the Euclidean algorithm
    have h₁₅ : gcd (2 * a + 3) (2 * a + 2) = gcd (2 * a + 3) 1 := by
      have h₁₆ : 2 * a + 2 = 2 * a + 3 - 1 := by ring
      rw [h₁₆]
      simp [gcd_comm]
      <;> simp_all [gcd_comm]
      <;> omega
    rw [h₁₅]
    -- The gcd of any number and 1 is 1
    have h₁₇ : gcd (2 * a + 3) 1 = 1 := by
      simp [gcd_comm]
    rw [h₁₇]
  exact h₁₂
