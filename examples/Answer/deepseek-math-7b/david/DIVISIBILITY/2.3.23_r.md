theorem gcd_dvd_of_dvd_mul {a b c : ℕ} (h : a ∣ b * c) : a ∣ Nat.gcd a b * Nat.gcd a c := by
  have h₁ : a ∣ Nat.gcd a b * Nat.gcd a c := by
    have h₂ : a ∣ b * c := h
    have h₃ : a ∣ Nat.gcd a b * Nat.gcd a c := by
      -- Use the fact that if a divides bc, then a divides the gcd of a and b times the gcd of a and c.
      apply Nat.dvd_trans h₂
      -- Show that bc divides the gcd of a and b times the gcd of a and c.
      rw [Nat.dvd_iff_mod_eq_zero]
      have h₄ : Nat.gcd a b * Nat.gcd a c ∣ b * c := by
        -- Use the property of gcd that gcd(a, b) * gcd(a, c) divides both b and c.
        have h₅ : Nat.gcd a b ∣ a := Nat.gcd_dvd_left a b
        have h₆ : Nat.gcd a c ∣ a := Nat.gcd_dvd_left a c
        have h₇ : Nat.gcd a b ∣ b := Nat.gcd_dvd_right a b
        have h₈ : Nat.gcd a c ∣ c := Nat.gcd_dvd_right a c
        -- Use the fact that if a number divides two others, it divides their product.
        exact Nat.mul_dvd_mul h₇ h₈
      simp_all [Nat.mod_eq_zero_of_dvd]
    exact h₃
  exact h₁
