theorem unique_q_r_2b_le_r_lt_3b (a b : ℤ) (hb : b ≠ 0) : ∃! (q r : ℤ), a = q * b + r ∧ 2 * b ≤ r ∧ r < 3 * b := by
  use (a / b)
  use (a % b)
  constructor
  · have h₁ : a = b * (a / b) + (a % b) := by rw [Int.ediv_add_emod]
    have h₂ : 2 * b ≤ a % b := by
      have h₃ : 0 ≤ a % b := by apply Int.emod_nonneg a (by intro h; apply hb; omega)
      have h₄ : a % b ≥ 2 * b := by
        have h₅ : a % b ≥ 0 := by apply Int.emod_nonneg a (by intro h; apply hb; omega)
        have h₆ : a % b < b := by apply Int.emod_lt_of_pos; omega
        omega
      omega
    have h₃ : a % b < 3 * b := by
      have h₄ : a % b < b := by apply Int.emod_lt_of_pos; omega
      omega
    exact ⟨by nlinarith, h₂, h₃⟩
  -- We need to show uniqueness, i.e., if there exist q' and r' such that a = q' * b + r' and 2 * b ≤ r' < 3 * b, then q' = q and r' = r.
  rintro q' r' ⟨h₁, h₂, h₃⟩
  have h₄ : q' = a / b := by
    have h₅ : q' * b + r' = a := by linarith
    have h₆ : q' * b + r' = b * (a / b) + (a % b) := by linarith
    have h₇ : q' = a / b := by
      apply Eq.symm
      apply Int.ediv_eq_of_eq_mul_right (by omega)
      linarith
    exact h₇
  have h₅ : r' = a % b := by
    have h₆ : q' * b + r' = a := by linarith
    have h₇ : q' * b + r' = b * (a / b) + (a % b) := by linarith
    have h₈ : r' = a % b := by
      apply Eq.symm
      apply Int.emod_eq_of_lt
      <;> omega
    exact h₈
  exact ⟨h₄, h₅⟩
