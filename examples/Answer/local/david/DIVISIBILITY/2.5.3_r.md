5 = 1 * 3 + 2
3 = 1 * 2 + 1

1 = 3 - 1 * 2
1 = 3 - 1 * (5 - 1 * 3)
1 = 2 * 3 - 1 * 5
1 = 2 * (18 - 1 * 5) - 1 * 5
