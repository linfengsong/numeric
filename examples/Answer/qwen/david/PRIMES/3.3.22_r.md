n^2 \equiv -3 \pmod{p} \quad \text{and} \quad (n+1)^2 \equiv -3 \pmod{p}

$$
n \equiv -\frac{1}{2} \pmod{p}

$$
\left(-\frac{1}{2}\right)^2 \equiv -3 \pmod{p} \Rightarrow \frac{1}{4} \equiv -3 \pmod{p}
